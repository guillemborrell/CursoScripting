def fib2(n): 
    """Devuelve una lista con la serie de 
       Fibonacci hasta n."""
    result = []
    (a, b) = (0, 1)
    while b < n:
        result.append(b)
        (a, b) = (b, a+b)
    return result
\end{lstlisting}
}
